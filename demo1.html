HTML Markup for an image


<div class="img-wrapper">
    <img src="path/to/minified/image.jpg" class="lazy" data-src="path/to/fullsize/image.jpg" alt="">
    <div class="loader"></div>
</div>

Some basic CSS


img {
    width: 100%;
    margin: 15px 0; 
}

.img-wrapper {
    position: relative; 
}

img.loaded + .loader {
    display: none; 
}
    
.lazy {
    filter: blur(6px);
    transition: 1s all ease; 
}
    
.lazy.loaded {
    filter: blur(0px); 
}

.loader {
    width: 60px;
    height: 60px;
    background-color: #fff;
    position: absolute;
    top: 50%;
    left: 50%;
    margin: -30px 0 0 -30px;
    border-radius: 100%;
    animation: pulse 0.7s infinite ease-in-out; 
}
    
@keyframes pulse {
    0% {
        transform: scale(0); 
    }
    100% {
        transform: scale(1);
        opacity: 0; 
    } 
}

Javascript


lazzzy.init({
    selector: '.lazy',
    allowLoadingOutsideViewport: false,
    classAfterLoading: 'loaded'
});
